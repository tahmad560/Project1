Structure in C Programming

Definition

In C, a structure is a user-defined data type that allows grouping variables of different types together under a single name.

Syntax

struct StructureName {
    dataType member1;
    dataType member2;
    // ... more members
};

Example

Hereâ€™s an example of defining and using a structure:

#include 

struct Student {
    int id;
    char name[50];
    float marks;
};

int main() {
    struct Student s1 = {1, "Alice", 85.5};

    printf("ID: %d\n", s1.id);
    printf("Name: %s\n", s1.name);
    printf("Marks: %.2f\n", s1.marks);

    return 0;
}

Other Ways to Input Data in Structures

1. Using scanf for Individual Members

#include 

struct Student {
    int id;
    char name[50];
    float marks;
};

int main() {
    struct Student s1;

    printf("Enter ID: ");
    scanf("%d", &s1.id);

    printf("Enter Name: ");
    scanf(" %[^"]", s1.name);

    printf("Enter Marks: ");
    scanf("%f", &s1.marks);

    printf("\nStudent Details:\n");
    printf("ID: %d\n", s1.id);
    printf("Name: %s\n", s1.name);
    printf("Marks: %.2f\n", s1.marks);

    return 0;
}

2. Using a Function

#include 

struct Student {
    int id;
    char name[50];
    float marks;
};

void inputStudent(struct Student *s) {
    printf("Enter ID: ");
    scanf("%d", &s->id);

    printf("Enter Name: ");
    scanf(" %[^"]", s->name);

    printf("Enter Marks: ");
    scanf("%f", &s->marks);
}

int main() {
    struct Student s1;

    printf("Enter details for the student:\n");
    inputStudent(&s1);

    printf("\nStudent Details:\n");
    printf("ID: %d\n", s1.id);
    printf("Name: %s\n", s1.name);
    printf("Marks: %.2f\n", s1.marks);

    return 0;
}

3. Using Loops for Arrays of Structures

#include 

struct Student {
    int id;
    char name[50];
    float marks;
};

int main() {
    int n;
    printf("Enter the number of students: ");
    scanf("%d", &n);

    struct Student students[n];

    for (int i = 0; i < n; i++) {
        printf("\nEnter details for student %d:\n", i + 1);
        printf("Enter ID: ");
        scanf("%d", &students[i].id);
        printf("Enter Name: ");
        scanf(" %[^"]", students[i].name);
        printf("Enter Marks: ");
        scanf("%f", &students[i].marks);
    }

    printf("\nStudent Details:\n");
    for (int i = 0; i < n; i++) {
        printf("ID: %d, Name: %s, Marks: %.2f\n", students[i].id, students[i].name, students[i].marks);
    }

    return 0;
}
